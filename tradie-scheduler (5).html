Google Sheets setup
Connect your live data
This app reads and writes jobs from a Google Sheet — so every manager sees the same data in real time from any device.
Go to sheets.google.com and create a new blank spreadsheet.
Name the first sheet Jobs and the second sheet Workers.
In the Jobs sheet, add these headers in row 1 exactly:
id | name | scope | start | end | labour | notes | status
In the Workers sheet, add headers: id | name | color — workers will sync here automatically when added.
Click Extensions → Apps Script, paste the web app script below, deploy as a web app (anyone can access), and copy the URL.
Paste the deployed web app URL below and tap Save.
// Paste this in Apps Script — replaces everything: function doGet(e) { var p = e.parameter; var callback = p.callback; var result; if (p.method === 'post' && p.data) { // write path — data comes in via GET param var payload = JSON.parse(decodeURIComponent(p.data)); var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(payload.sheet || 'Jobs'); if (payload.action === 'set') { var headers = sheet.getRange(1,1,1,sheet.getLastColumn()).getValues()[0]; var lastRow = sheet.getLastRow(); if (lastRow > 1) sheet.deleteRows(2, lastRow - 1); payload.rows.forEach(function(r) { sheet.appendRow(headers.map(function(h){ return r[h] !== undefined ? r[h] : ''; })); }); } result = { ok: true }; } else { // read path var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheetName = p.sheet || 'Jobs'; var sheet = ss.getSheetByName(sheetName); var data = sheet.getDataRange().getValues(); var headers = data[0]; result = data.slice(1).filter(function(r){ return r.some(function(c){ return c !== ''; }); }).map(function(r){ var obj = {}; headers.forEach(function(h,i){ obj[h] = r[i]; }); return obj; }); } var json = JSON.stringify(result); if (callback) { return ContentService .createTextOutput(callback + '(' + json + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); }
